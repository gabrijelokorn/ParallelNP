93 mutable struct TwoPhaseDefUseMap
94
95 This struct is intended as a memory- and GC-pressure-efficient mechanism
96 for incrementally computing def-use maps. The idea is that the def-use map
97 is constructed into two passes over the IR. In the first, we simply count the
98 the number of uses, computing the number of uses for each def as well as the
99 total number of uses. In the second pass, we actually fill in the def-use
100 information.
101
102 The idea is that either of these two phases can be combined with other useful
103 work that needs to scan the instruction stream anyway, while avoiding the
104 significant allocation pressure of e.g. allocating an array for every SSA value
105 or attempting to dynamically move things around as new uses are discovered.
106
107 The def-use map is presented as a vector of vectors. For every def, indexing
108 into the map will return a vector of uses.
109 """

function compute_trycatch(code::Vector{Any}, ip::BitSet)
367 # The goal initially is to record the frame like this for the state at exit:
368 # 1: (enter 3) # == 0
369 # 3: (expr) # == 1
370 # 3: (leave %1) # == 1
371 # 4: (expr) # == 0
372 # then we can find all `try`s by walking backwards from :enter statements,
373 # and all `catch`es by looking at the statement after the :enter
374 n = length(code)
375 empty!(ip)
376 ip.offset = 0 # for _bits_findnext
377 push!(ip, n + 1)
378 1 (0 %)
1 (100 %) samples spent calling fill
handler_at = fill((0, 0), n)
379 handlers = TryCatchFrame[]
380
381 # start from all :enter statements and record the location of the try
382 for pc = 1:n
383 stmt = code[pc]
384 if isa(stmt, EnterNode)
385 l = stmt.catch_dest
386 push!(handlers, TryCatchFrame(Bottom, isdefined(stmt, :scope) ? Bottom : nothing, pc))
387 handler_id = length(handlers)
388 handler_at[pc + 1] = (handler_id, 0)
389 push!(ip, pc + 1)
390 if l != 0
391 handler_at[l] = (0, handler_id)
392 push!(ip, l)
393 end
394 end
395 end
396
397 # now forward those marks to all :leave statements
398 while true
399 # make progress on the active ip set
400 pc = _bits_findnext(ip.bits, 0)::Int
401 pc > n && break
402 while true # inner loop optimizes the common case where it can run straight from pc to pc + 1
403 pc = pc + 1 # next program-counter (after executing instruction)
404 delete!(ip, pc)
405 cur_stacks = handler_at[pc]
406 @assert cur_stacks != (0, 0) "unbalanced try/catch"
407 stmt = code[pc]
408 if isa(stmt, GotoNode)
409 pc = stmt.label
410 elseif isa(stmt, GotoIfNot)
411 l = stmt.dest::Int
412 if handler_at[l] != cur_stacks
413 @assert handler_at[l][1] == 0 || handler_at[l][1] == cur_stacks[1] "unbalanced try/catch"
414 handler_at[l] = cur_stacks
415 push!(ip, l)
416 end
417 elseif isa(stmt, ReturnNode)
418 @assert !isdefined(stmt, :val) || cur_stacks[1] == 0 "unbalanced try/catch"
419 break
420 elseif isa(stmt, EnterNode)
421 l = stmt.catch_dest
422 # We assigned a handler number above. Here we just merge that
423 # with out current handler information.
424 if l != 0
425 handler_at[l] = (cur_stacks[1], handler_at[l][2])
426 end
427 cur_stacks = (handler_at[pc][1], cur_stacks[2])
428 elseif isa(stmt, Expr)
429 head = stmt.head
430 if head === :leave
431 l = 0
432 for j = 1:length(stmt.args)
433 arg = stmt.args[j]
434 if arg === nothing
435 continue
436 else
437 enter_stmt = code[(arg::SSAValue).id]
438 if enter_stmt === nothing
439 continue
440 end
441 @assert isa(enter_stmt, EnterNode) "malformed :leave"
442 end
443 l += 1
444 end
445 cur_hand = cur_stacks[1]
446 for i = 1:l
447 cur_hand = handler_at[handlers[cur_hand].enter_idx][1]
448 end

455 end

520 """
521 constrains_param(var::TypeVar, sig, covariant::Bool, type_constrains::Bool)
522
523 Check if `var` will be constrained to have a definite value
524 in any concrete leaftype subtype of `sig`.
525
526 It is used as a helper to determine whether type intersection is guaranteed to be able to
527 find a value for a particular type parameter.
528 A necessary condition for type intersection to not assign a parameter is that it only
529 appears in a `Union[All]` and during subtyping some other union component (that does not
530 constrain the type parameter) is selected.
531
532 The `type_constrains` flag determines whether Type{T} is considered to be constraining
533 `T`. This is not true in general, because of the existence of types with free type
534 parameters, however, some callers would like to ignore this corner case.
535 """
536 function constrains_param(var::TypeVar, @nospecialize(typ), covariant::Bool, type_constrains::Bool=false)
537 typ === var && return true
538 while typ isa UnionAll
539 covariant && constrains_param(var, typ.var.ub, covariant, type_constrains) && return true
540 # typ.var.lb doesn't constrain var
541 typ = typ.body
542 end
543 if typ isa Union
544 # for unions, verify that both options would constrain var
545 ba = constrains_param(var, typ.a, covariant, type_constrains)
546 bb = constrains_param(var, typ.b, covariant, type_constrains)
547 (ba && bb) && return true
548 elseif typ isa DataType
549 # return true if any param constrains var
550 fc = length(typ.parameters)
551 if fc > 0
552 if typ.name === Tuple.name
553 # vararg tuple needs special handling
554 for i in 1:(fc - 1)
555 p = typ.parameters[i]
556 constrains_param(var, p, covariant, type_constrains) && return true
557 end
558 lastp = typ.parameters[fc]
559 vararg = unwrap_unionall(lastp)
560 if vararg isa Core.TypeofVararg && isdefined(vararg, :N)
561 constrains_param(var, vararg.N, covariant, type_constrains) && return true
562 # T = vararg.parameters[1] doesn't constrain var
563 else
564 constrains_param(var, lastp, covariant, type_constrains) && return true
565 end
566 else
567 if typ.name === typename(Type) && typ.parameters[1] === var && var.ub === Any
568 # Types with free type parameters are <: Type cause the typevar
569 # to be unconstrained because Type{T} with free typevars is illegal
570 return type_constrains
571 end
572 for i in 1:fc
573 p = typ.parameters[i]
574 constrains_param(var, p, false, type_constrains) && return true
575 end
576 end
577 end
578 end
579 return false

624 end
625 end
626 end
627 ub = unwraptv_ub(v)
628 if has_free_typevars(ub)
629 ub = Any
630 end
631 lb = unwraptv_lb(v)
632 if has_free_typevars(lb)
633 lb = Bottom
634 end
635 if Any === ub && lb === Bottom
636 ty = Any
637 else
638 tv = TypeVar(v.name, lb, ub)
639 ty = UnionAll(tv, Type{tv})
640 end
641 @label ty_computed
642 undef = !(let sig=sig
643 # if the specialized signature `linfo.specTypes` doesn't contain any free
644 # type variables, we can use it for a more accurate analysis of whether `v`
645 # is constrained or not, otherwise we should use `def.sig` which always
646 # doesn't contain any free type variables
647 if !has_free_typevars(linfo.specTypes)
648 sig = linfo.specTypes
649 end
650 @assert !has_free_typevars(sig)
651 constrains_param(v, sig, #=covariant=#true)
652 end)
653 elseif isvarargtype(v)
654 # if this parameter came from `func(..., ::Vararg{T,v})`,
655 # so the type is known to be `Int`
656 ty = Int
657 undef = false
658 else
659 ty = Const(v)
660 undef = false
661 end
662 sptypes[i] = VarState(ty, undef)
663 end
664 return sptypes
665 end

872 Iterate through all callers of the given `AbsIntState` in the abstract interpretation stack
