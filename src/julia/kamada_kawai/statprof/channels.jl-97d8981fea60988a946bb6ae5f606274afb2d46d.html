13 """
14 Channel{T=Any}(size::Int=0)
15
16 Constructs a `Channel` with an internal buffer that can hold a maximum of `size` objects
17 of type `T`.
18 [`put!`](@ref) calls on a full channel block until an object is removed with [`take!`](@ref).
19
20 `Channel(0)` constructs an unbuffered channel. `put!` blocks until a matching `take!` is called.
21 And vice-versa.
22
23 Other constructors:
24
25 * `Channel()`: default constructor, equivalent to `Channel{Any}(0)`
26 * `Channel(Inf)`: equivalent to `Channel{Any}(typemax(Int))`
27 * `Channel(sz)`: equivalent to `Channel{Any}(sz)`
28
29 !!! compat "Julia 1.3"
30 The default constructor `Channel()` and default `size=0` were added in Julia 1.3.
31 """

193 close(c::Channel[, excp::Exception])
194
195 Close a channel. An exception (optionally given by `excp`), is thrown by:
196
197 * [`put!`](@ref) on a closed channel.
198 * [`take!`](@ref) and [`fetch`](@ref) on an empty, closed channel.

200 close(c::Channel) = close(c, closed_exception()) # nospecialize on default arg seems to confuse makedocs
201 function close(c::Channel, @nospecialize(excp::Exception))
202 lock(c)
203 try
204 c.excp = excp
205 @atomic :release c.state = :closed
206 notify_error(c.cond_take, excp)
207 notify_error(c.cond_wait, excp)
208 notify_error(c.cond_put, excp)
209 finally
210 unlock(c)
211 end
212 nothing
213 end

215 # Use acquire here to pair with release store in `close`, so that subsequent `isready` calls
216 # are forced to see `isready == true` if they see `isopen == false`. This means users must
217 # call `isopen` before `isready` if you are using the race-y APIs (or call `iterate`, which
218 # does this right for you).
219 isopen(c::Channel) = ((@atomic :acquire c.state) === :open)

238 bind(chnl::Channel, task::Task)
239
240 Associate the lifetime of `chnl` with a task.
241 `Channel` `chnl` is automatically closed when the task terminates.
242 Any uncaught exception in the task is propagated to all waiters on `chnl`.
243
244 The `chnl` object can be explicitly closed independent of task termination.
245 Terminating tasks have no effect on already closed `Channel` objects.
246
247 When a channel is bound to multiple tasks, the first task to terminate will
248 close the channel. When multiple channels are bound to the same task,
249 termination of the task will close all of the bound channels.
250
251 # Examples

253 julia> c = Channel(0);
254
255 julia> task = @async foreach(i->put!(c, i), 1:4);

296 """
297 channeled_tasks(n::Int, funcs...; ctypes=fill(Any,n), csizes=fill(0,n))
298
299 A convenience method to create `n` channels and bind them to tasks started
300 from the provided functions in a single call. Each `func` must accept `n` arguments
301 which are the created channels. Channel types and sizes may be specified via
302 keyword arguments `ctypes` and `csizes` respectively. If unspecified, all channels are
303 of type `Channel{Any}(0)`.
304
305 Returns a tuple, `(Array{Channel}, Array{Task})`, of the created channels and tasks.
306 """
307 function channeled_tasks(n::Int, funcs...; ctypes=fill(Any,n), csizes=fill(0,n))
308 @assert length(csizes) == n
309 @assert length(ctypes) == n
310
311 chnls = map(i -> Channel{ctypes[i]}(csizes[i]), 1:n)
312 tasks = Task[ Task(() -> f(chnls...)) for f in funcs ]
313
314 # bind all tasks to all channels and schedule them
315 foreach(t -> foreach(c -> bind(c, t), chnls), tasks)
316 foreach(schedule, tasks)
317 yield() # Allow scheduled tasks to run
318
319 return (chnls, tasks)
320 end

374 try
375 # Increment channel n_avail eagerly (before push!) to count data in the
376 # buffer as well as offers from tasks which are blocked in wait().
377 _increment_n_avail(c, 1)
378 while length(c.data) == c.sz_max

388 # Decrement the available items if this task had an exception before pushing the

390 did_buffer || _increment_n_avail(c, -1)
