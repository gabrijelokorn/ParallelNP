5 """
6 GenericMemory{kind::Symbol, T, addrspace=Core.CPU} <: DenseVector{T}
7
8 Fixed-size [`DenseVector{T}`](@ref DenseVector).
9
10 `kind` can currently be either `:not_atomic` or `:atomic`. For details on what `:atomic` implies, see [`AtomicMemory`](@ref)
11
12 `addrspace` can currently only be set to Core.CPU. It is designed to to permit extension by other systems
13 such as GPUs, which might define values such as:
14 ```
15 module CUDA
16 const Generic = bitcast(Core.AddrSpace{CUDA}, 0)
17 const Global = bitcast(Core.AddrSpace{CUDA}, 1)
18 end
19 ```
20 The exact semantics of these other addrspaces is defined by the specific backend, but will error if the user is attempting to access these on the CPU.
21
22 !!! compat "Julia 1.11"
23 This type requires Julia 1.11 or later.

37 """
38 AtomicMemory{T} == GenericMemory{:atomic, T, Core.CPU}
39
40 Fixed-size [`DenseVector{T}`](@ref DenseVector).
41 Access to its any of its elements is performed atomically (with `:monotonic` ordering).
42 Setting any of the elements must be accomplished using the `@atomic` macro and explicitly specifying ordering.
43
44 !!! warning
45 Each element is independently atomic when accessed, and cannot be set non-atomically.
46 Currently the `@atomic` macro and higher level interface have not been completed,
47 but the building blocks for a future implementation are the internal intrinsics
48 `Core.memoryrefget`, `Core.memoryrefset!`, `Core.memoryref_isassigned`, `Core.memoryrefswap!`,
49 `Core.memoryrefmodify!`, and `Core.memoryrefreplace!`.
50
51 For details, see [Atomic Operations](@ref man-atomic-operations)

74 function _unsetindex!(A::MemoryRef{T}) where T
75 @_terminates_locally_meta
76 @_propagate_inbounds_meta
77 @inline
78 @boundscheck memoryref(A, 1)
79 mem = A.mem
80 MemT = typeof(mem)
81 arrayelem = datatype_arrayelem(MemT)
82 elsz = datatype_layoutsize(MemT)
83 isboxed = 1; isunion = 2
84 t = @_gc_preserve_begin mem
85 p = Ptr{Ptr{Cvoid}}(@inbounds pointer(A))
86 if arrayelem == isboxed
87 Intrinsics.atomic_pointerset(p, C_NULL, :monotonic)
88 elseif arrayelem != isunion
89 if !datatype_pointerfree(T::DataType)
90 for j = 1:Core.sizeof(Ptr{Cvoid}):elsz
91 # XXX: this violates memory ordering, since it writes more than one C_NULL to each
92 Intrinsics.atomic_pointerset(p + j - 1, C_NULL, :monotonic)
93 end
94 end
95 end
96 @_gc_preserve_end t
97 return A
98 end

134 function unsafe_copyto!(dest::Memory, doffs, src::Memory, soffs, n)
135 @_terminates_locally_meta
136 n == 0 && return dest
137 # use pointer math to determine if they are deemed to alias
138 destp = pointer(dest, doffs)
139 srcp = pointer(src, soffs)
140 endp = pointer(src, soffs + n - 1)
141 @inbounds if destp < srcp || destp > endp
142 for i = 1:n
143 if isassigned(src, soffs + i - 1)
144 dest[doffs + i - 1] = src[soffs + i - 1]
145 else
146 _unsetindex!(dest, doffs + i - 1)
147 end
148 end
149 else
150 for i = n:-1:1
151 if isassigned(src, soffs + i - 1)
152 dest[doffs + i - 1] = src[soffs + i - 1]
153 else
154 _unsetindex!(dest, doffs + i - 1)
155 end
156 end
157 end
158 return dest
159 end

266 # use memcmp for cmp on byte arrays
267 function cmp(a::Memory{UInt8}, b::Memory{UInt8})
268 ta = @_gc_preserve_begin a
269 tb = @_gc_preserve_begin b
270 pa = unsafe_convert(Ptr{Cvoid}, a)
271 pb = unsafe_convert(Ptr{Cvoid}, b)
272 c = memcmp(pa, pb, min(length(a),length(b)))
273 @_gc_preserve_end ta
274 @_gc_preserve_end tb
275 return c < 0 ? -1 : c > 0 ? +1 : cmp(length(a),length(b))
276 end
