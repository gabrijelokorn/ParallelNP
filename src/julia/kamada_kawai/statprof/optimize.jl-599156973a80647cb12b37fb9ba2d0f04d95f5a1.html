308 # TODO: We're duplicating analysis from inference here.
309 isa(stmt, PiNode) && return (true, true, true)
310 isa(stmt, PhiNode) && return (true, true, true)
311 isa(stmt, ReturnNode) && return (true, false, true)
312 isa(stmt, EnterNode) && return (true, false, true)
313 isa(stmt, GotoNode) && return (true, false, true)
314 isa(stmt, GotoIfNot) && return (true, false, (, argextype(stmt.cond, src), Bool))
315 if isa(stmt, GlobalRef)
316 nothrow = consistent = isdefinedconst_globalref(stmt)
317 return (consistent, nothrow, nothrow)
318 elseif isa(stmt, Expr)
319 (; head, args) = stmt
320 if head === :static_parameter
321 # if we aren't certain enough about the type, it might be an UndefVarError at runtime
322 sptypes = isa(src, IRCode) ? src.sptypes : src.ir.sptypes
323 nothrow = !sptypes[args[1]::Int].undef
324 return (true, nothrow, nothrow)
325 end
326 if head === :call
327 f = argextype(args[1], src)
328 f = singleton_type(f)
329 f === nothing && return (false, false, false)
330 if f === UnionAll
331 # TODO: This is a weird special case - should be determined in inference
332 argtypes = Any[argextype(args[arg], src) for arg in 2:length(args)]
333 nothrow = _builtin_nothrow(, f, argtypes, rt)
334 return (true, nothrow, nothrow)
335 end
336 if f === Intrinsics.cglobal || f === Intrinsics.llvmcall
337 # TODO: these are not yet linearized
338 return (false, false, false)
339 end

341 # Needs to be handled in inlining to look at the callee effects
342 f === Core._apply_iterate && return (false, false, false)
343 argtypes = Any[argextype(args[arg], src) for arg in 2:length(args)]
344 effects = builtin_effects(, f, argtypes, rt)
345 consistent = is_consistent(effects)
346 effect_free = is_effect_free(effects)
347 nothrow = is_nothrow(effects)
348 removable = effect_free & nothrow
349 return (consistent, removable, nothrow)
350 elseif head === :new
351 return new_expr_effect_flags(, args, src)
352 elseif head === :foreigncall

354 argextype(x, src)
355 end
356 consistent = is_consistent(effects)
357 effect_free = is_effect_free(effects)
358 nothrow = is_nothrow(effects)
359 removable = effect_free & nothrow
360 return (consistent, removable, nothrow)
361 elseif head === :new_opaque_closure
362 length(args) < 4 && return (false, false, false)
363 typ = argextype(args[1], src)
364 typ, isexact = instanceof_tfunc(typ, true)
365 isexact || return (false, false, false)
366 (, typ, Tuple) || return (false, false, false)
367 rt_lb = argextype(args[2], src)
368 rt_ub = argextype(args[3], src)
369 source = argextype(args[4], src)
370 if !((, rt_lb, Type) && (, rt_ub, Type) && (, source, Method))
371 return (false, false, false)
372 end
373 return (false, true, true)
374 elseif head === :inbounds
375 return (true, true, true)
376 elseif head === :boundscheck || head === :isdefined || head === :the_exception || head === :copyast
377 return (false, true, true)
378 else
379 # e.g. :loopinfo
380 return (false, false, false)
381 end
382 end
383 isa(stmt, SlotNumber) && error("unexpected IR elements")
384 return (true, true, true)

468 ir::IRCode, caller::InferenceResult)
469 (; src, linfo) = opt
470 (; def, specTypes) = linfo

472 force_noinline = is_declared_noinline(src)
473
474 # compute inlining and other related optimizations
475 result = caller.result
476 @assert !(result isa LimitedAccuracy)
477 result = widenslotwrapper(result)
478
479 opt.ir = ir

481 # determine and cache inlineability
482 if !force_noinline
483 sig = unwrap_unionall(specTypes)
484 if !(isa(sig, DataType) && sig.name === Tuple.name)
485 force_noinline = true
486 end
487 if !is_declared_inline(src) && result === Bottom
488 force_noinline = true
489 end
490 end

492 set_inlineable!(src, false)
493 elseif isa(def, Method)
494 if is_declared_inline(src) && isdispatchtuple(specTypes)
495 # obey @inline declaration if a dispatch barrier would not help
496 set_inlineable!(src, true)
497 else
498 # compute the cost (size) of inlining this code
499 params = OptimizationParams(interp)
500 cost_threshold = default = params.inline_cost_threshold
501 if (optimizer_lattice(interp), result, Tuple) && !isconcretetype(widenconst(result))

503 end
504 # if the method is declared as `@inline`, increase the cost threshold 20x

507 end

510 name = def.name
511 if name === :iterate || name === :unsafe_convert || name === :cconvert
512 cost_threshold += 4*default
513 end
514 end

516 end

1169 sv.insert_coverage && (labelchangemap[oldidx + 1] += 1)
1170 end
1171 if blockchangemap === nothing
1172 blockchangemap = fill(0, length(sv.cfg.blocks))
1173 end
1174 blockchangemap[block] += 1
1175 idx += 1
1176 end
1177 oldidx = last(sv.cfg.blocks[block].stmts)
1178 end
1179 end
1180 idx += 1
1181 oldidx += 1
1182 end
1183
1184 if ssachangemap !== nothing && labelchangemap !== nothing
1185 renumber_ir_elements!(code, ssachangemap, labelchangemap)
1186 end
1187 if blockchangemap !== nothing
1188 renumber_cfg_stmts!(sv.cfg, blockchangemap)
1189 end
1190
1191 for i = 1:length(code)
1192 code[i] = process_meta!(meta, code[i])
1193 end
1194 strip_trailing_junk!(ci, sv.cfg, code, stmtinfo)
1195 types = Any[]
1196 stmts = InstructionStream(code, types, stmtinfo, codelocs, ssaflags)

1198 # types of call arguments only once `slot2reg` converts this `IRCode` to the SSA form
1199 # and eliminates slots (see below)
1200 argtypes = sv.slottypes
1201 return IRCode(stmts, sv.cfg, linetable, argtypes, meta, sv.sptypes)

1226 ## Computing the cost of a function body
1227
1228 # saturating sum (inputs are non-negative), prevents overflow with typemax(Int) below
1229 plus_saturate(x::Int, y::Int) = max(x, y, x+y)

1231 # known return type
1232 isknowntype(@nospecialize T) = (T === Union{}) || isa(T, Const) || isconcretetype(widenconst(T))

1234 function statement_cost(ex::Expr, line::Int, src::Union{CodeInfo, IRCode}, sptypes::Vector{VarState},
1235 params::OptimizationParams, error_path::Bool = false)
1236 #=const=# UNKNOWN_CALL_COST = 20
1237 head = ex.head
1238 if is_meta_expr_head(head)
1239 return 0
1240 elseif head === :call
1241 farg = ex.args[1]
1242 ftyp = argextype(farg, src, sptypes)
1243 if ftyp === IntrinsicFunction && farg isa SSAValue
1244 # if this comes from code that was already inlined into another function,
1245 # Consts have been widened. try to recover in simple cases.
1246 farg = isa(src, CodeInfo) ? src.code[farg.id] : src[farg][:stmt]
1247 if isa(farg, GlobalRef) || isa(farg, QuoteNode) || isa(farg, IntrinsicFunction) || isexpr(farg, :static_parameter)
1248 ftyp = argextype(farg, src, sptypes)
1249 end
1250 end
1251 f = singleton_type(ftyp)
1252 if isa(f, IntrinsicFunction)
1253 iidx = Int(reinterpret(Int32, f::IntrinsicFunction)) + 1
1254 if isassigned(T_IFUNC, iidx)
1255 minarg, maxarg, = T_IFUNC[iidx]
1256 nargs = length(ex.args)
1257 if minarg + 1 <= nargs <= maxarg + 1
1258 # With mostly constant arguments, all Intrinsics tend to become very cheap
1259 # and are likely to combine with the operations around them,
1260 # so reduce their cost by half.
1261 cost = T_IFUNC_COST[iidx]
1262 if cost == 0 || nargs < 3 ||
1263 (f === Intrinsics.cglobal || f === Intrinsics.llvmcall) # these hold malformed IR, so argextype will crash on them
1264 return cost
1265 end
1266 aty2 = widenconditional(argextype(ex.args[2], src, sptypes))
1267 nconst = Int(aty2 isa Const)
1268 for i = 3:nargs
1269 aty = widenconditional(argextype(ex.args[i], src, sptypes))
1270 if widenconst(aty) != widenconst(aty2)
1271 nconst = 0
1272 break
1273 end
1274 nconst += aty isa Const
1275 end
1276 if nconst + 2 >= nargs
1277 cost = (cost - 1) 2
1278 end
1279 return cost
1280 end
1281 end
1282 # unknown/unhandled intrinsic: hopefully the caller gets a slightly better answer after the inlining
1283 return UNKNOWN_CALL_COST
1284 end
1285 if isa(f, Builtin) && f !== invoke
1286 # The efficiency of operations like a[i] and s.b
1287 # depend strongly on whether the result can be
1288 # inferred, so check the type of ex
1289 if f === Core.getfield || f === Core.tuple || f === Core.getglobal
1290 # we might like to penalize non-inferrability, but
1291 # tuple iteration/destructuring makes that impossible
1292 # return plus_saturate(argcost, isknowntype(extyp) ? 1 : params.inline_nonleaf_penalty)
1293 return 0
1294 elseif (f === Core.memoryrefget || f === Core.memoryref_isassigned) && length(ex.args) >= 3
1295 atyp = argextype(ex.args[2], src, sptypes)
1296 return isknowntype(atyp) ? 1 : error_path ? params.inline_error_path_cost : params.inline_nonleaf_penalty
1297 elseif f === Core.memoryrefset! && length(ex.args) >= 3
1298 atyp = argextype(ex.args[2], src, sptypes)
1299 return isknowntype(atyp) ? 5 : error_path ? params.inline_error_path_cost : params.inline_nonleaf_penalty
1300 elseif f === typeassert && isconstType(widenconst(argextype(ex.args[3], src, sptypes)))
1301 return 1
1302 end
1303 fidx = find_tfunc(f)
1304 if fidx === nothing
1305 # unknown/unhandled builtin
1306 # Use the generic cost of a direct function call
1307 return UNKNOWN_CALL_COST
1308 end
1309 return T_FFUNC_COST[fidx]
1310 end
1311 extyp = line == -1 ? Any : argextype(SSAValue(line), src, sptypes)
1312 if extyp === Union{}
1313 return 0
1314 end
1315 return error_path ? params.inline_error_path_cost : params.inline_nonleaf_penalty
1316 elseif head === :foreigncall
1317 foreigncall = ex.args[1]
1318 if foreigncall isa QuoteNode && foreigncall.value === :jl_string_ptr
1319 return 1
1320 end
1321 return 20
1322 elseif head === :invoke || head === :invoke_modify
1323 # Calls whose "return type" is Union{} do not actually return:
1324 # they are errors. Since these are not part of the typical
1325 # run-time of the function, we omit them from
1326 # consideration. This way, non-inlined error branches do not
1327 # prevent inlining.
1328 extyp = line == -1 ? Any : argextype(SSAValue(line), src, sptypes)
1329 return extyp === Union{} ? 0 : UNKNOWN_CALL_COST
1330 elseif head === :(=)
1331 if ex.args[1] isa GlobalRef
1332 cost = UNKNOWN_CALL_COST
1333 else
1334 cost = 0
1335 end
1336 a = ex.args[2]
1337 if a isa Expr
1338 cost = plus_saturate(cost, statement_cost(a, -1, src, sptypes, params, error_path))
1339 end
1340 return cost
1341 elseif head === :copyast
1342 return 100
1343 end
1344 return 0
1345 end

1347 function statement_or_branch_cost(@nospecialize(stmt), line::Int, src::Union{CodeInfo, IRCode}, sptypes::Vector{VarState},
1348 params::OptimizationParams)
1349 thiscost = 0
1350 dst(tgt) = isa(src, IRCode) ? first(src.cfg.blocks[tgt].stmts) : tgt
1351 if stmt isa Expr
1352 thiscost = statement_cost(stmt, line, src, sptypes, params,
1353 is_stmt_throw_block(isa(src, IRCode) ? src.stmts.flag[line] : src.ssaflags[line]))::Int
1354 elseif stmt isa GotoNode
1355 # loops are generally always expensive
1356 # but assume that forward jumps are already counted for from
1357 # summing the cost of the not-taken branch
1358 thiscost = dst(stmt.label) < line ? 40 : 0
1359 elseif stmt isa GotoIfNot
1360 thiscost = dst(stmt.dest) < line ? 40 : 0
1361 elseif stmt isa EnterNode
1362 # try/catch is a couple function calls,
1363 # but don't inline functions with try/catch
1364 # since these aren't usually performance-sensitive functions,
1365 # and llvm is more likely to miscompile them when these functions get large
1366 thiscost = typemax(Int)
1367 end
1368 return thiscost
1369 end

1371 function inline_cost(ir::IRCode, params::OptimizationParams, cost_threshold::Int)
1372 bodycost = 0
1373 for i = 1:length(ir.stmts)
1374 stmt = ir[SSAValue(i)][:stmt]
1375 thiscost = statement_or_branch_cost(stmt, i, ir, ir.sptypes, params)
1376 bodycost = plus_saturate(bodycost, thiscost)
1377 if bodycost > cost_threshold
1378 return MAX_INLINE_COST
1379 end
1380 end
1381 return inline_cost_clamp(bodycost)
1382 end

1402 function cumsum_ssamap!(ssachangemap::Vector{Int})
1403 any_change = false
1404 rel_change = 0
1405 for i = 1:length(ssachangemap)
1406 val = ssachangemap[i]
1407 any_change |= val 0
1408 rel_change += val
1409 if val == -1
1410 # Keep a marker that this statement was deleted
1411 ssachangemap[i] = typemin(Int)
1412 else
1413 ssachangemap[i] = rel_change
1414 end
1415 end
1416 return any_change
1417 end

1419 function renumber_ir_elements!(body::Vector{Any}, ssachangemap::Vector{Int}, labelchangemap::Vector{Int})
1420 any_change = cumsum_ssamap!(labelchangemap)
1421 if ssachangemap !== labelchangemap
1422 any_change |= cumsum_ssamap!(ssachangemap)
1423 end
1424 any_change || return
1425 for i = 1:length(body)
1426 el = body[i]
1427 if isa(el, GotoNode)
1428 body[i] = GotoNode(el.label + labelchangemap[el.label])
1429 elseif isa(el, GotoIfNot)
1430 cond = el.cond
1431 if isa(cond, SSAValue)
1432 cond = SSAValue(cond.id + ssachangemap[cond.id])
1433 end
1434 was_deleted = labelchangemap[el.dest] == typemin(Int)
1435 body[i] = was_deleted ? cond : GotoIfNot(cond, el.dest + labelchangemap[el.dest])
1436 elseif isa(el, ReturnNode)
1437 if isdefined(el, :val)
1438 val = el.val
1439 if isa(val, SSAValue)
1440 body[i] = ReturnNode(SSAValue(val.id + ssachangemap[val.id]))
1441 end
1442 end
1443 elseif isa(el, SSAValue)
1444 body[i] = SSAValue(el.id + ssachangemap[el.id])
1445 elseif isa(el, PhiNode)
1446 i = 1
1447 edges = el.edges
1448 values = el.values
1449 while i <= length(edges)
1450 was_deleted = ssachangemap[edges[i]] == typemin(Int)
1451 if was_deleted
1452 deleteat!(edges, i)
1453 deleteat!(values, i)
1454 else
1455 edges[i] += ssachangemap[edges[i]]
1456 val = values[i]
1457 if isa(val, SSAValue)
1458 values[i] = SSAValue(val.id + ssachangemap[val.id])
1459 end
1460 i += 1
1461 end
1462 end
1463 elseif isa(el, EnterNode)
1464 tgt = el.catch_dest
1465 if tgt != 0
1466 was_deleted = labelchangemap[tgt] == typemin(Int)
1467 if was_deleted
1468 @assert !isdefined(el, :scope)
1469 body[i] = nothing
1470 else
1471 if isdefined(el, :scope) && isa(el.scope, SSAValue)
1472 body[i] = EnterNode(tgt + labelchangemap[tgt], SSAValue(el.scope.id + ssachangemap[el.scope.id]))
1473 else
1474 body[i] = EnterNode(el, tgt + labelchangemap[tgt])
1475 end
1476 end
1477 end
1478 elseif isa(el, Expr)
1479 if el.head === :(=) && el.args[2] isa Expr
1480 el = el.args[2]::Expr
1481 end
1482 if !is_meta_expr_head(el.head)
1483 args = el.args
1484 for i = 1:length(args)
1485 el = args[i]
1486 if isa(el, SSAValue)
1487 args[i] = SSAValue(el.id + ssachangemap[el.id])
1488 end
1489 end
1490 end
1491 end
1492 end
1493 end
