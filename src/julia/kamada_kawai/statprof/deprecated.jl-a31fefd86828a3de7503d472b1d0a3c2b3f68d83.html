15 """
16 @deprecate old new [export_old=true]
17
18 Deprecate method `old` and specify the replacement call `new`, defining a new method `old`
19 with the specified signature in the process.
20
21 To prevent `old` from being exported, set `export_old` to `false`.
22
23 See also [`Base.depwarn()`](@ref).
24
25 !!! compat "Julia 1.5"
26 As of Julia 1.5, functions defined by `@deprecate` do not print warning when `julia`
27 is run without the `--depwarn=yes` flag set, as the default value of `--depwarn` option
28 is `no`. The warnings are printed from tests run by `Pkg.test()`.
29
30 # Examples
31 ```jldoctest
32 julia> @deprecate old(x) new(x)
33 old (generic function with 1 method)
34
35 julia> @deprecate old(x) new(x) false
36 old (generic function with 1 method)
37 ```
38
39 Calls to `@deprecate` without explicit type-annotations will define
40 deprecated methods accepting any number of positional and keyword
41 arguments of type `Any`.
42
43 !!! compat "Julia 1.9"
44 Keyword arguments are forwarded when there is no explicit type
45 annotation as of Julia 1.9. For older versions, you can manually
46 forward positional and keyword arguments by doing `@deprecate
47 old(args...; kwargs...) new(args...; kwargs...)`.
48
49 To restrict deprecation to a specific signature, annotate the
50 arguments of `old`. For example,
51 ```jldoctest; filter = r"@ .*"a
52 julia> new(x::Int) = x;
53
54 julia> new(x::Float64) = 2x;
55
56 julia> @deprecate old(x::Int) new(x);
57
58 julia> methods(old)
59 # 1 method for generic function "old" from Main:
60 [1] old(x::Int64)
61 @ deprecated.jl:94
62 ```
63 will define and deprecate a method `old(x::Int)` that mirrors `new(x::Int)` but will not
64 define nor deprecate the method `old(x::Float64)`.
65 """

121 """
122 Base.depwarn(msg::String, funcsym::Symbol; force=false)
123
124 Print `msg` as a deprecation warning. The symbol `funcsym` should be the name
125 of the calling function, which is used to ensure that the deprecation warning is
126 only printed the first time for each call place. Set `force=true` to force the
127 warning to always be shown, even if Julia was started with `--depwarn=no` (the
128 default).
129
130 See also [`@deprecate`](@ref).
131
132 # Examples

136
137 1 + 1
138 end
139 ```

179 firstcaller(bt::Vector, ::Nothing) = Ptr{Cvoid}(0), StackTraces.UNKNOWN
180 firstcaller(bt::Vector, funcsym::Symbol) = firstcaller(bt, (funcsym,))
181 function firstcaller(bt::Vector, funcsyms)
182 # Identify the calling line
183 found = false
184 for ip in bt
185 lkups = StackTraces.lookup(ip)
186 for lkup in lkups
187 if lkup == StackTraces.UNKNOWN || lkup.from_c
188 continue
189 end
190 if found
191 return ip, lkup
192 end
193 found = lkup.func in funcsyms
194 # look for constructor type name
195 if !found
196 li = lkup.linfo
197 if li isa Core.MethodInstance
198 def = li.def
199 found = def isa Method && def.name in funcsyms
200 end
201 end
202 end
203 end
204 return C_NULL, StackTraces.UNKNOWN
205 end

361 """
362 resize_nthreads!(A, copyvalue=A[1])
363
364 Resize the array `A` to length [`nthreads()`](@ref). Any new
365 elements that are allocated are initialized to `deepcopy(copyvalue)`,
366 where `copyvalue` defaults to `A[1]`.
367
368 This is typically used to allocate per-thread variables, and
369 should be called in `__init__` if `A` is a global constant.
370
371 !!! warning
372
373 This function is deprecated, since as of Julia v1.9 the number of
374 threads can change at run time. Instead, per-thread state should be
375 created as needed based on the thread id of the caller.
376 """
377 function resize_nthreads!(A::AbstractVector, copyvalue=A[1])
378 nthr = nthreads()
379 nold = length(A)
380 resize!(A, nthr)
381 for i = nold+1:nthr
382 A[i] = deepcopy(copyvalue)
383 end
384 return A
385 end
386 end
